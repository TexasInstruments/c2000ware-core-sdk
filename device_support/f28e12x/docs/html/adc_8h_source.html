599 static inline void
600 ADC_setupSOC(uint32_t base, ADC_SOCNumber socNumber, ADC_Trigger trigger,
601  ADC_Channel channel, uint32_t sampleWindow)
602 {
603  uint32_t ctlRegAddr, mask;
604  uint32_t ACQPS;
605 
606  //
607  // Check the arguments.
608  //
609  ASSERT(ADC_isBaseValid(base));
610 
611  mask = (ADC_SOC0CTL_CHSEL_M | ADC_SOC0CTL_TRIGSEL_M | ADC_SOC0CTL_ACQPS_M);
612 
613  if (sampleWindow > 448UL){
614  ACQPS = (3UL << 6UL) | (((sampleWindow - 448UL) >> 4UL) - 1UL);
615  }
616  else if (sampleWindow > 192UL){
617  ACQPS = (2UL << 6UL) | (((sampleWindow - 192UL) >> 2UL) - 1UL);
618 
619  }
620  else if (sampleWindow > 64UL){
621  ACQPS = (1UL << 6UL) | (((sampleWindow - 64UL) >> 1UL) - 1UL);
622 
623  }
624  else{
625  ACQPS = sampleWindow - 1UL;
626  }
627  //
628  // Calculate address for the SOC control register.
629  //
630  ctlRegAddr = base + ADC_SOCxCTL_OFFSET_BASE + ((uint32_t)socNumber * 2U);
631 
632  //
633  // Set the configuration of the specified SOC.
634  //
635  EALLOW;
636 
637  HWREG(ctlRegAddr) = (HWREG(ctlRegAddr) & ~(mask)) |
638  ((uint32_t)channel << ADC_SOC0CTL_CHSEL_S) |
639  ((uint32_t)trigger << ADC_SOC0CTL_TRIGSEL_S) |
640  ACQPS;
641 
642  EDIS;
643 }
