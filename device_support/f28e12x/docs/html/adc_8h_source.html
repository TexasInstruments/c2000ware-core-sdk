589 //
628 //
629 //*****************************************************************************
630 static inline void
631 ADC_setupSOC(uint32_t base, ADC_SOCNumber socNumber, ADC_Trigger trigger,
632  ADC_Channel channel, uint32_t sampleWindow)
633 {
634  uint32_t ctlRegAddr, mask;
635  uint32_t ACQPS;
636 
637  //
638  // Check the arguments.
639  //
640  ASSERT(ADC_isBaseValid(base));
641 
642  mask = (ADC_SOC0CTL_CHSEL_M | ADC_SOC0CTL_TRIGSEL_M | ADC_SOC0CTL_ACQPS_M);
643 
644  if (sampleWindow > 448UL){
645  ACQPS = (3UL << 6UL) | (((sampleWindow - 448UL) >> 4UL) - 1UL);
646  }
647  else if (sampleWindow > 192UL){
648  ACQPS = (2UL << 6UL) | (((sampleWindow - 192UL) >> 2UL) - 1UL);
649 
650  }
651  else if (sampleWindow > 64UL){
652  ACQPS = (1UL << 6UL) | (((sampleWindow - 64UL) >> 1UL) - 1UL);
653 
654  }
655  else{
656  ACQPS = sampleWindow - 1UL;
657  }
658  //
659  // Calculate address for the SOC control register.
660  //
661  ctlRegAddr = base + ADC_SOCxCTL_OFFSET_BASE + ((uint32_t)socNumber * 2U);
662 
663  //
664  // Set the configuration of the specified SOC.
665  //
666  EALLOW;
667 
668  HWREG(ctlRegAddr) = (HWREG(ctlRegAddr) & ~(mask)) |
669  ((uint32_t)channel << ADC_SOC0CTL_CHSEL_S) |
670  ((uint32_t)trigger << ADC_SOC0CTL_TRIGSEL_S) |
671  ACQPS;
672 
673  EDIS;
674 }
