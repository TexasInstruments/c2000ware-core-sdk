412 static inline void ASysCtl_selectCMPHNMuxValue(uint16_t select, uint16_t value)
413 {
414  ASSERT(select <= 0x7FU);
415  ASSERT(value <= 1U);
416 
417  EALLOW;
418 
419  //
420  // Write a select to the mux select bit.
421  //
422  if(value == 0U)
423  {
424  HWREGH(ANALOGSUBSYS_BASE + ASYSCTL_O_CMPHNMXSEL) &= ~select;
425  }
426  else if(value == 1U)
427  {
428  HWREGH(ANALOGSUBSYS_BASE + ASYSCTL_O_CMPHNMXSEL) |= select;
429  }
430  else
431  {
432  //Do nothing. Not a valid value
433  }
434 
435  EDIS;
436 }
