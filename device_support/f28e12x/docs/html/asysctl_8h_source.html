407 static inline void ASysCtl_selectCMPHNMuxValue(uint16_t select, uint16_t value)
408 {
409  ASSERT(select <= 0x7FU);
410  ASSERT(value <= 1U);
411 
412  EALLOW;
413 
414  //
415  // Write a select to the mux select bit.
416  //
417  if(value == 0U)
418  {
419  HWREGH(ANALOGSUBSYS_BASE + ASYSCTL_O_CMPHNMXSEL) &= ~select;
420  }
421  else if(value == 1U)
422  {
423  HWREGH(ANALOGSUBSYS_BASE + ASYSCTL_O_CMPHNMXSEL) |= select;
424  }
425  else
426  {
427  //Do nothing. Not a valid value
428  }
429 
430  EDIS;
431 }

850 {
851  ASSERT((pin == 0U) || (pin == 23U) || (pin == 25U) ||
852  (pin == 32U) || (pin == 46U) || (pin == 103U));
853 
854  uint16_t mask =
855  (pin == 43U) ? ASYSCTL_GPIOINENACTRL_GPIO43 :
856  (pin == 44U) ? ASYSCTL_GPIOINENACTRL_GPIO44 :
857  (pin == 45U) ? ASYSCTL_GPIOINENACTRL_GPIO45 :
858  0U;
859 
860  EALLOW;
861  HWREGH(ANALOGSUBSYS_BASE + ASYSCTL_O_GPIOINENACTRL) |= mask;
