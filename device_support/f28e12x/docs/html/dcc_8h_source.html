805 static inline void
806 DCC_setCounterSeeds(uint32_t base, uint32_t counter0, uint32_t validCounter0,
807  uint32_t counter1)
808 {
809  //
810  // Check the arguments.
811  //
812  ASSERT(DCC_isBaseValid(base));
813  ASSERT(validCounter0 >= DCC_VALIDSEED_MIN);
814  ASSERT(counter0 >= DCC_COUNTSEED_MIN);
815  ASSERT(counter1 >= DCC_COUNTSEED_MIN);
816  ASSERT(validCounter0 <= DCC_VALIDSEED_MAX);
817  ASSERT(counter0 <= DCC_COUNTSEED_MAX);
818  ASSERT(counter1 <= DCC_COUNTSEED_MAX);
819 
820  EALLOW;
821 
822  //
823  // Set Counter 0 Seed
824  //
825  HWREGH(base + DCC_O_CNTSEED0) = counter0 & DCC_REG_WORD_MASK;
826  HWREGH(base + DCC_O_CNTSEED0 + 2U) = (HWREGH(base + DCC_O_CNTSEED0 + 2U) &
827  DCC_SEED_REG_MASK) |
828  ((uint32_t)(counter0 &
829  DCC_SEED_CNT_MASK) >> 16U);
830 
831  //
832  // Set Valid Duration Counter 0 Seed
833  //
834  HWREGH(base + DCC_O_VALIDSEED0) = validCounter0;
835 
836  //
837  // Set Counter 1 Seed
838  //
839  HWREGH(base + DCC_O_CNTSEED1) = counter1 & DCC_REG_WORD_MASK;
840  HWREGH(base + DCC_O_CNTSEED1 + 2U) = (HWREGH(base + DCC_O_CNTSEED1 + 2U) &
841  DCC_SEED_REG_MASK) |
842  ((uint32_t)(counter1 &
843  DCC_SEED_CNT_MASK) >> 16U);
844 
845  EDIS;
846 }
