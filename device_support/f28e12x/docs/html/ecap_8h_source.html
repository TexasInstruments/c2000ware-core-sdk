301 //*****************************************************************************
302 static inline void ECAP_setEventPrescaler(uint32_t base,
303  uint16_t preScalerValue)
304 {
305  ASSERT(ECAP_isBaseValid(base));
306 
307  ASSERT(preScalerValue < ECAP_MAX_PRESCALER_VALUE);
308 
309  EALLOW;
310 
311  //
312  // Write to PRESCALE bit
313  //
314  HWREGH(base + ECAP_O_ECCTL1) =
315  ((HWREGH(base + ECAP_O_ECCTL1) & (~ECAP_ECCTL1_PRESCALE_M)) |
316  (preScalerValue << ECAP_ECCTL1_PRESCALE_S));
317  EDIS;
318 }

320 //*****************************************************************************
321 //
336 //
337 //*****************************************************************************
338 static inline void ECAP_setEventPolarity(uint32_t base,
339  ECAP_Events event,
340  ECAP_EventPolarity polarity)
341 {
342 
343  uint16_t shift;
344 
345  ASSERT(ECAP_isBaseValid(base));
346 
347  shift = ((uint16_t)event) << 1U;
348 
349  EALLOW;
350 
351  //
352  // Write to CAP1POL, CAP2POL, CAP3POL or CAP4POL
353  //
354  HWREGH(base + ECAP_O_ECCTL1) =
355  (HWREGH(base + ECAP_O_ECCTL1) & ~(1U << shift)) |
356  ((uint16_t)polarity << shift);
357  EDIS;
358 }

451 static inline void ECAP_enableInterrupt(uint32_t base,
452  uint16_t intFlags)
453 {
454  ASSERT(ECAP_isBaseValid(base));
455  ASSERT((intFlags & ~(ECAP_ISR_SOURCE_CAPTURE_EVENT_1 |
456  ECAP_ISR_SOURCE_CAPTURE_EVENT_2 |
457  ECAP_ISR_SOURCE_CAPTURE_EVENT_3 |
458  ECAP_ISR_SOURCE_CAPTURE_EVENT_4 |
459  ECAP_ISR_SOURCE_COUNTER_OVERFLOW |
460  ECAP_ISR_SOURCE_COUNTER_PERIOD |
461  ECAP_ISR_SOURCE_COUNTER_COMPARE)) == 0U);
462 
463 
464  EALLOW;
465 
466  //
467  // Set bits in ECEINT register
468  //
469  HWREGH(base + ECAP_O_ECEINT) |= intFlags;
470  EDIS;
471 }
