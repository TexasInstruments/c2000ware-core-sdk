187 static inline void
188 Flash_setWaitstates(uint32_t ctrlBase, uint16_t waitstates)
189 {
190  //
191  // Check the arguments.
192  //
193  ASSERT(Flash_isCtrlBaseValid(ctrlBase));
194 
195  //
196  // waitstates is 4 bits wide.
197  //
198  ASSERT(waitstates <= 0xFU);
199 
200  EALLOW;
201  //
202  // Write flash read wait-state amount to appropriate register.
203  //
204  HWREG(ctrlBase + FLASH_O_FRDCNTL) =
205  (HWREG(ctrlBase + FLASH_O_FRDCNTL) &
206  ~(uint32_t)FLASH_FRDCNTL_RWAIT_M) |
207  ((uint32_t)waitstates << FLASH_FRDCNTL_RWAIT_S);
208  EDIS;
209 }
