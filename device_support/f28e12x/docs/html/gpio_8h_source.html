258 {

260  // Write the selected polarity to the appropriate register.

407 {
408  volatile uint32_t *gpioDataReg;

411  // Check the arguments.

417 
418  return((gpioDataReg[GPIO_GPxDAT_INDEX] >> (pin % 32U)) & (uint32_t)0x1U);

471 GPIO_writePin(uint32_t pin, uint32_t outVal)
472 {
473  volatile uint32_t *gpioDataReg;
474  uint32_t pinMask;
475 
476  //
477  // Check the arguments.
478  //
479  ASSERT(GPIO_isPinValid(pin));
480 
481  gpioDataReg = (uint32_t *)((uintptr_t)GPIODATA_BASE) +
482  ((pin / 32U) * GPIO_DATA_REGS_STEP);
483 
484  pinMask = (uint32_t)1U << (pin % 32U);
485 
486  if(outVal == 0U)
487  {
488  gpioDataReg[GPIO_GPxCLEAR_INDEX] = pinMask;
489  }
490  else
491  {
492  gpioDataReg[GPIO_GPxSET_INDEX] = pinMask;
493  }
494 }

530  pinMask = (uint32_t)1U << (pin % 32U);
