342 static inline void
343 SCI_lockAutobaud(uint32_t base)
344 {
345  //
346  // Check the arguments.
347  //
348  ASSERT(SCI_isBaseValid(base));
349 
350  //
351  // Prime the baud register
352  //
353  HWREGH(base + SCI_O_HBAUD) = 0x0U;
354  HWREGH(base + SCI_O_LBAUD) = 0x1U;
355 
356  //
357  // Prepare for autobaud detection.
358  // Set the CDC bit to enable autobaud detection and clear the ABD bit.
359  //
360  HWREGH(base + SCI_O_FFCT) |= SCI_FFCT_CDC;
361  HWREGH(base + SCI_O_FFCT) |= SCI_FFCT_ABDCLR;
362 
363  //
364  // Wait until we correctly read an 'A' or 'a' and lock
365  //
366  while((HWREGH(base + SCI_O_FFCT) & SCI_FFCT_ABD) != SCI_FFCT_ABD)
367  {
368  }
369 
370  //
371  // After autobaud lock, clear the ABD and CDC bits
372  //
373  HWREGH(base + SCI_O_FFCT) |= SCI_FFCT_ABDCLR;
374  HWREGH(base + SCI_O_FFCT) &= ~SCI_FFCT_CDC;
375 }

476 static inline void
477 SCI_getConfig(uint32_t base, uint32_t lspclkHz, uint32_t *baud,
478  uint32_t *config)
479 {
480  //
481  // Check the arguments.
482  //
483  ASSERT(SCI_isBaseValid(base));
484 
485  //
486  // Compute the baud rate.
487  //
488  *baud = lspclkHz /
489  ((1U + (((uint32_t)HWREGH(base + SCI_O_HBAUD) << 8U) |
490  HWREGH(base + SCI_O_LBAUD))) * 8U);
491 
492  //
493  // Get the parity, data length, and number of stop bits.
494  //
495  *config = (uint32_t)HWREGH(base + SCI_O_CCR) & (SCI_CONFIG_PAR_MASK |
496  SCI_CONFIG_STOP_MASK |
497  SCI_CONFIG_WLEN_MASK);
498 }
