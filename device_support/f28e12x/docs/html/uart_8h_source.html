336 static inline void
337 UART_setParityMode(uint32_t base, uint32_t parity)
338 {
339  //
340  // Check the arguments.
341  //
342  ASSERT(UART_isBaseValid(base));
343  ASSERT((parity == UART_CONFIG_PAR_NONE) ||
344  (parity == UART_CONFIG_PAR_EVEN) ||
345  (parity == UART_CONFIG_PAR_ODD) ||
346  (parity == UART_CONFIG_PAR_ONE) ||
347  (parity == UART_CONFIG_PAR_ZERO));
348 
349  //
350  // Set the parity mode.
351  //
352  HWREG(base + UART_O_LCRH) = ((HWREG(base + UART_O_LCRH) &
353  ~(UART_LCRH_SPS | UART_LCRH_EPS |
354  UART_LCRH_PEN)) | parity);
355 }

449 static inline void
450 UART_getFIFOLevel(uint32_t base, uint32_t *txLevel,
451  uint32_t *rxLevel)
452 {
453  uint32_t temp;
454 
455  //
456  // Check the arguments.
457  //
458  ASSERT(UART_isBaseValid(base));
459 
460  //
461  // Read the FIFO level register.
462  //
463  temp = HWREG(base + UART_O_IFLS);
464 
465  //
466  // Extract the transmit and receive FIFO levels.
467  //
468  *txLevel = temp & UART_IFLS_TXIFLSEL_M;
469  *rxLevel = temp & UART_IFLS_RXIFLSEL_M;
470 }
