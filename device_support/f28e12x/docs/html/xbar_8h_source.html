503 //
504 //*****************************************************************************
505 static inline void
506 XBAR_setOutputLatchMode(uint32_t base, XBAR_OutputNum output, bool enable)
507 {
508  //
509  // Check the arguments.
510  //
511  ASSERT(XBAR_isBaseValid(base));
512 
513  EALLOW;
514 
515  //
516  // Set or clear the latch setting bit based on the enable parameter.
517  //
518  if(enable)
519  {
520  HWREGH(base + XBAR_O_OUTPUTLATCHENABLE) |=
521  0x1U << ((uint16_t)output / 2U);
522  }
523  else
524  {
525  HWREGH(base + XBAR_O_OUTPUTLATCHENABLE) &=
526  ~(0x1U << ((uint16_t)output / 2U));
527  }
528 
529  EDIS;
530 }

821

858 //*****************************************************************************
859 //
870 //
871 //*****************************************************************************
872 static inline void
873 XBAR_lockOutput(uint32_t base)
874 {
875  //
876  // Check the arguments.
877  //
878  ASSERT(XBAR_isBaseValid(base));
879 
880  //
881  // Lock the Output X-BAR with the OUTPUTLOCK register.
882  // Write key 0x5A5A to the KEY bits and 1 to LOCK bit.
883  //
884  EALLOW;
885 
886  HWREG(base + XBAR_O_OUTPUTLOCK) =
887  ((uint32_t)0x5A5A << XBAR_OUTPUTLOCK_KEY_S) |
888  (uint32_t)XBAR_OUTPUTLOCK_LOCK;
