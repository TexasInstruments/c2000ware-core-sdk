1024 //
1044 //
1045 //*****************************************************************************
1046 static inline void CLB_configLocalInputMux(uint32_t base, CLB_Inputs inID,
1047  CLB_LocalInputMux localMuxCfg)
1048 {
1049  uint16_t shiftVal;
1050  uint32_t miscShiftVal, inputMuxSel;
1051 
1052  ASSERT(CLB_isBaseValid(base));
1053 
1054  //
1055  // Each local input has a 5-bit field in this register so need to calculate
1056  // shift amount accordingly.
1057  //
1058  shiftVal = (uint16_t)inID * CLB_LCL_MUX_SEL_1_LCL_MUX_SEL_IN_1_S;
1059 
1060  //
1061  // If the local inputs are selected then only 5-bits are used in this
1062  // register but if miscellaneous inputs are selected then another 1-bit is
1063  // used along with the above 5. So we need to calculate this accordingly.
1064  //
1065 
1066  //
1067  // Each miscellaneous input has a 1-bit field in this register so
1068  // need to calculate shift amount accordingly.
1069  //
1070  miscShiftVal = (uint32_t)inID + CLB_LCL_MUX_SEL_MISC_INPUT_SEL_S;
1071 
1072  //
1073  // Check if input < input4
1074  //
1075  if(inID < CLB_IN4)
1076  {
1077  //
1078  // Each miscellaneous input has an extra 1-bit to be set
1079  // apart from the local input bits
1080  //
1081  inputMuxSel =
1082  ((((uint32_t)localMuxCfg & CLB_LCL_MUX_SEL_1_LCL_MUX_SEL_IN_0_M) <<
1083  shiftVal) |
1084  ((((uint32_t)localMuxCfg & CLB_LCL_MUX_SEL_MISC_INPUT_SEL_M) >>
1085  CLB_LCL_MUX_SEL_1_LCL_MUX_SEL_IN_1_S) << miscShiftVal));
1086 
1087  //
1088  //Update the local / misc input mux selections
1089  //
1090  HWREG(base + CLB_LOGICCTL + CLB_O_LCL_MUX_SEL_1) =
1091  (HWREG(base + CLB_LOGICCTL + CLB_O_LCL_MUX_SEL_1) &
1092  ~(((uint32_t)CLB_LCL_MUX_SEL_1_LCL_MUX_SEL_IN_0_M << shiftVal) |
1093  (CLB_LCL_MUX_SEL_MISC_INPUT_SEL_BITM << miscShiftVal))) |
1094  inputMuxSel;
1095  }
1096  else
1097  {
1098  //
1099  //Adjust the previously calculated shifts for the inputs >3
1100  //
1101  shiftVal -= 4U * CLB_LCL_MUX_SEL_2_LCL_MUX_SEL_IN_5_S;
1102  miscShiftVal -= 4U;
1103 
1104  //
1105  // Each miscellaneous input has an extra 1-bit to be set
1106  // apart from the local input bits
1107  //
1108  inputMuxSel =
1109  ((((uint32_t)localMuxCfg & CLB_LCL_MUX_SEL_2_LCL_MUX_SEL_IN_4_M) <<
1110  shiftVal) |
1111  ((((uint32_t)localMuxCfg & CLB_LCL_MUX_SEL_MISC_INPUT_SEL_M) >>
1112  CLB_LCL_MUX_SEL_2_LCL_MUX_SEL_IN_5_S) << miscShiftVal));
1113 
1114  HWREG(base + CLB_LOGICCTL + CLB_O_LCL_MUX_SEL_2) =
1115  (HWREG(base + CLB_LOGICCTL + CLB_O_LCL_MUX_SEL_2) &
1116  ~(((uint32_t)CLB_LCL_MUX_SEL_2_LCL_MUX_SEL_IN_4_M << shiftVal) |
1117  (CLB_LCL_MUX_SEL_MISC_INPUT_SEL_BITM << miscShiftVal))) |
1118  inputMuxSel;
1119 
1120  }
1121 }
