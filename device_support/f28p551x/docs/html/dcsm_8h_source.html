357 //*****************************************************************************
358 //
364 //
365 //*****************************************************************************
366 static inline DCSM_SecurityStatus
367 DCSM_getZone1CSMSecurityStatus(void)
368 {
369  uint32_t status;
370  DCSM_SecurityStatus returnStatus;
371  status = HWREG(DCSM_Z1_BASE + DCSM_O_Z1_CR);
372 
373  //
374  // if ARMED bit is set and UNSECURED bit is set then CSM is unsecured.
375  // Else it is secure.
376  //
377  if(((status & DCSM_Z1_CR_ARMED) != 0U) &&
378  ((status & DCSM_Z1_CR_UNSECURE) != 0U))
379  {
380  returnStatus = DCSM_STATUS_UNSECURE;
381  }
382  else if((status & DCSM_Z1_CR_ALLONE) == DCSM_Z1_CR_ALLONE)
383  {
384  returnStatus = DCSM_STATUS_BLOCKED;
385  }
386  else if((status & DCSM_Z1_CR_ALLZERO) == DCSM_Z1_CR_ALLZERO)
387  {
388  returnStatus = DCSM_STATUS_LOCKED;
389  }
390  else
391  {
392  returnStatus = DCSM_STATUS_SECURE;
393  }
394 
395  return(returnStatus);
396 }

523 //*****************************************************************************
524 //
535 //
536 //*****************************************************************************
537 static inline DCSM_MemoryStatus
538 DCSM_getFlashSectorZone(DCSM_Sector sector)
539 {
540  uint32_t sectStat;
541  uint16_t shift;
542 
543  //
544  // Get the Sector status register for the specific bank
545  //
546  if(sector <= DCSM_BANK0_SECTOR_127_96)
547  {
548  sectStat = HWREG(DCSMCOMMON_BASE + DCSM_O_SECTSTAT1);
549  shift = (uint16_t)sector * 2U;
550  }
551  else if(sector <= DCSM_BANK2_SECTOR_127_96)
552  {
553  sectStat = HWREG(DCSMCOMMON_BASE + DCSM_O_SECTSTAT2);
554  shift = ((uint16_t)sector & 0xFU) * (uint16_t)2U;
555  }
556  else
557  {
558  sectStat = HWREG(DCSMCOMMON_BASE + DCSM_O_SECTSTAT3);
559  shift = ((uint16_t)sector & 0xFU) * (uint16_t)2U;
560  }
561 
562  //
563  //Read the SECTSTAT register for the specific Flash Sector.
564  //
565  return((DCSM_MemoryStatus)((uint16_t)((sectStat >> shift) & 0x3U)));
566 }

682 //*****************************************************************************
683 static inline uint32_t
684 DCSM_getZone1OTPSecureLockStatus(DCSM_OTPLock lockType)
685 {
686  uint32_t status, returnStatus;
687  status = HWREG(DCSM_Z1_BASE + DCSM_O_Z1_OTPSECLOCK);
688 
689  //
690  // Reflects the state of the OTP Sec LOCK feature.
691  //
692  if(lockType == DCSM_OTPSECLOCK_JTAG)
693  {
694  returnStatus = (status & DCSM_Z1_OTPSECLOCK_JTAGLOCK_M) >>
695  DCSM_Z1_OTPSECLOCK_JTAGLOCK_S;
696  }
697  else if(lockType == DCSM_OTPSECLOCK_CRCLOCK)
698  {
699  returnStatus = (status & DCSM_Z1_OTPSECLOCK_CRCLOCK_M) >>
700  DCSM_Z1_OTPSECLOCK_CRCLOCK_S;
701  }
702  else if(lockType == DCSM_OTPSECLOCK_PSWDLOCK)
703  {
704  returnStatus = (status & DCSM_Z1_OTPSECLOCK_PSWDLOCK_M) >>
705  DCSM_Z1_OTPSECLOCK_PSWDLOCK_S;
706  }
707  else
708  {
709  returnStatus = (uint32_t)DCSM_OTPSECLOCK_INVALID;
710  }
711 
712  return(returnStatus);
713 }
