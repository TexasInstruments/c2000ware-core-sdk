395 static inline void ECAP_setEventPrescaler(uint32_t base,
396  uint16_t preScalerValue)
397 {
398  ASSERT(ECAP_isBaseValid(base));
399 
400  ASSERT(preScalerValue < ECAP_MAX_PRESCALER_VALUE);
401 
402  EALLOW;
403 
404  //
405  // Write to PRESCALE bit
406  //
407  HWREGH(base + ECAP_O_ECCTL1) =
408  ((HWREGH(base + ECAP_O_ECCTL1) & (~ECAP_ECCTL1_PRESCALE_M)) |
409  (preScalerValue << ECAP_ECCTL1_PRESCALE_S));
410  EDIS;
411 }

431 static inline void ECAP_setEventPolarity(uint32_t base,
432  ECAP_Events event,
433  ECAP_EventPolarity polarity)
434 {
435 
436  uint16_t shift;
437 
438  ASSERT(ECAP_isBaseValid(base));
439 
440  shift = ((uint16_t)event) << 1U;
441 
442  EALLOW;
443 
444  //
445  // Write to CAP1POL, CAP2POL, CAP3POL or CAP4POL
446  //
447  HWREGH(base + ECAP_O_ECCTL1) =
448  (HWREGH(base + ECAP_O_ECCTL1) & ~(1U << shift)) |
449  ((uint16_t)polarity << shift);
450  EDIS;
451 }
