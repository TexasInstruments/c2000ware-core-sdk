459 //
460 //*****************************************************************************
461 static inline void
462 EPG_selectEPGOutput(uint32_t base, EPG_OUT gPinNum,
463  EPG_OutputSelect gPinOutSel)
464 {
465  uint32_t regValue, bitFieldLoc, bitFieldVal;
466 
467  ASSERT(EPG_isBaseValid(base));
468  ASSERT((uint16_t)gPinNum <= 7U);
469 
470  bitFieldLoc = 1UL << ((uint16_t)gPinNum + EPG_GCTL0_EPGOUTSEL_S);
471  bitFieldVal = (uint32_t)gPinOutSel << ((uint16_t)gPinNum +
472  EPG_GCTL0_EPGOUTSEL_S);
473 
474  regValue = (HWREG(base + EPG_O_GCTL0) & (~bitFieldLoc));
475  regValue |= bitFieldVal;
476 
477  HWREG((base + EPG_O_GCTL0)) = regValue;
478 }
