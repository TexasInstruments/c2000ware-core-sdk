319 static inline void
320 MemCfg_setCLAMemType(uint32_t ramSections, MemCfg_CLAMemoryType claMemType)
321 {
322  //
323  // Check the arguments.
324  //
325  ASSERT((ramSections & MEMCFG_SECT_TYPE_MASK) == MEMCFG_SECT_TYPE_LS);
326 
327  //
328  // Write the CLA memory configuration to the appropriate register. Either
329  // set or clear the bit that determines the function of the RAM section as
330  // it relates to the CLA.
331  //
332  EALLOW;
333 
334  if(claMemType == MEMCFG_CLA_MEM_PROGRAM)
335  {
336  //
337  // Program memory
338  //
339  HWREG(MEMCFG_BASE + MEMCFG_O_LSXCLAPGM) |= ramSections;
340  }
341  else
342  {
343  //
344  // Data memory
345  //
346  HWREG(MEMCFG_BASE + MEMCFG_O_LSXCLAPGM) &= ~ramSections;
347  }
348 
349  EDIS;
350 }
