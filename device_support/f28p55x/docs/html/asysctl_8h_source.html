374  ~refMask) | reference;

650 //*****************************************************************************
651 //
665 //
666 //*****************************************************************************
667 static inline void ASysCtl_selectCMPHNMuxValue(uint16_t select, uint16_t value)
668 {
669  ASSERT(select <= 0x7FU);
670  ASSERT(value <= 1U);
671 
672  EALLOW;
673 
674  //
675  // Write a select to the mux select bit.
676  //
677  if(value == 0U)
678  {
679  HWREGH(ANALOGSUBSYS_BASE + ASYSCTL_O_CMPHNMXSEL) &= ~select;
680  }
681  else if(value == 1U)
682  {
683  HWREGH(ANALOGSUBSYS_BASE + ASYSCTL_O_CMPHNMXSEL) |= select;
684  }
685  else
686  {
687  //Do nothing. Not a valid value
688  }
689 
690  EDIS;
691 }
