340  (HWREGH(ANALOGSUBSYS_BASE + ASYSCTL_O_ANAREFPCTL) &
341  ~refMask) | reference;
342 
343  EDIS;
344 }

588 //*****************************************************************************
589 //
603 //
604 //*****************************************************************************
605 static inline void ASysCtl_selectCMPHNMuxValue(uint16_t select, uint16_t value)
606 {
607  ASSERT(select <= 0x7FU);
608  ASSERT(value <= 1U);
609 
610  EALLOW;
611 
612  //
613  // Write a select to the mux select bit.
614  //
615  if(value == 0U)
616  {
617  HWREGH(ANALOGSUBSYS_BASE + ASYSCTL_O_CMPHNMXSEL) &= ~select;
618  }
619  else if(value == 1U)
620  {
621  HWREGH(ANALOGSUBSYS_BASE + ASYSCTL_O_CMPHNMXSEL) |= select;
622  }
623  else
624  {
625  //Do nothing. Not a valid value
626  }
627 
628  EDIS;
629 }
