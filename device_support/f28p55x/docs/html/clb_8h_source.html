1007 //*****************************************************************************
1008 //
1028 //
1029 //*****************************************************************************
1030 static inline void CLB_configLocalInputMux(uint32_t base, CLB_Inputs inID,
1031  CLB_LocalInputMux localMuxCfg)
1032 {
1033  uint16_t shiftVal;
1034  uint32_t miscShiftVal, inputMuxSel;
1035 
1036  ASSERT(CLB_isBaseValid(base));
1037 
1038  //
1039  // Each local input has a 5-bit field in this register so need to calculate
1040  // shift amount accordingly.
1041  //
1042  shiftVal = (uint16_t)inID * CLB_LCL_MUX_SEL_1_LCL_MUX_SEL_IN_1_S;
1043 
1044  //
1045  // If the local inputs are selected then only 5-bits are used in this
1046  // register but if miscellaneous inputs are selected then another 1-bit is
1047  // used along with the above 5. So we need to calculate this accordingly.
1048  //
1049 
1050  //
1051  // Each miscellaneous input has a 1-bit field in this register so
1052  // need to calculate shift amount accordingly.
1053  //
1054  miscShiftVal = (uint32_t)inID + CLB_LCL_MUX_SEL_MISC_INPUT_SEL_S;
1055 
1056  //
1057  // Check if input < input4
1058  //
1059  if(inID < CLB_IN4)
1060  {
1061  //
1062  // Each miscellaneous input has an extra 1-bit to be set
1063  // apart from the local input bits
1064  //
1065  inputMuxSel =
1066  ((((uint32_t)localMuxCfg & CLB_LCL_MUX_SEL_1_LCL_MUX_SEL_IN_0_M) <<
1067  shiftVal) |
1068  ((((uint32_t)localMuxCfg & CLB_LCL_MUX_SEL_MISC_INPUT_SEL_M) >>
1069  CLB_LCL_MUX_SEL_1_LCL_MUX_SEL_IN_1_S) << miscShiftVal));
1070 
1071  //
1072  //Update the local / misc input mux selections
1073  //
1074  HWREG(base + CLB_LOGICCTL + CLB_O_LCL_MUX_SEL_1) =
1075  (HWREG(base + CLB_LOGICCTL + CLB_O_LCL_MUX_SEL_1) &
1076  ~(((uint32_t)CLB_LCL_MUX_SEL_1_LCL_MUX_SEL_IN_0_M << shiftVal) |
1077  (CLB_LCL_MUX_SEL_MISC_INPUT_SEL_BITM << miscShiftVal))) |
1078  inputMuxSel;
1079  }
1080  else
1081  {
1082  //
1083  //Adjust the previously calculated shifts for the inputs >3
1084  //
1085  shiftVal -= 4U * CLB_LCL_MUX_SEL_2_LCL_MUX_SEL_IN_5_S;
1086  miscShiftVal -= 4U;
1087 
1088  //
1089  // Each miscellaneous input has an extra 1-bit to be set
1090  // apart from the local input bits
1091  //
1092  inputMuxSel =
1093  ((((uint32_t)localMuxCfg & CLB_LCL_MUX_SEL_2_LCL_MUX_SEL_IN_4_M) <<
1094  shiftVal) |
1095  ((((uint32_t)localMuxCfg & CLB_LCL_MUX_SEL_MISC_INPUT_SEL_M) >>
1096  CLB_LCL_MUX_SEL_2_LCL_MUX_SEL_IN_5_S) << miscShiftVal));
1097 
1098  HWREG(base + CLB_LOGICCTL + CLB_O_LCL_MUX_SEL_2) =
1099  (HWREG(base + CLB_LOGICCTL + CLB_O_LCL_MUX_SEL_2) &
1100  ~(((uint32_t)CLB_LCL_MUX_SEL_2_LCL_MUX_SEL_IN_4_M << shiftVal) |
1101  (CLB_LCL_MUX_SEL_MISC_INPUT_SEL_BITM << miscShiftVal))) |
1102  inputMuxSel;
1103 
1104  }
1105 }
