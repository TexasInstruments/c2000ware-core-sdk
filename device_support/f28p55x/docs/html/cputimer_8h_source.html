299  uint16_t tcrValue = 0;

301

304  //

306  HWREGH(base + CPUTIMER_O_TCR) = tcrValue | CPUTIMER_TCR_TRB;
307

431 static inline void CPUTimer_selectClockSource(uint32_t base,
432  CPUTimer_ClockSource source,
433  CPUTimer_Prescaler prescaler)
434 {
435  ASSERT(base == CPUTIMER2_BASE);
436 
437  //
438  // Set source and prescaler for CPU Timer 2
439  //
440  if(base == CPUTIMER2_BASE)
441  {
442  EALLOW;
443 
444  //
445  // Set Clock Source
446  //
447  HWREGH(CPUSYS_BASE + SYSCTL_O_TMR2CLKCTL) =
448  (HWREGH(CPUSYS_BASE + SYSCTL_O_TMR2CLKCTL) &
449  ~SYSCTL_TMR2CLKCTL_TMR2CLKSRCSEL_M) | (uint16_t)source;
450  SYSCTL_REGWRITE_DELAY;
451 
452  //
453  // Set Clock Prescaler
454  //
455  HWREGH(CPUSYS_BASE + SYSCTL_O_TMR2CLKCTL) =
456  (HWREGH(CPUSYS_BASE + SYSCTL_O_TMR2CLKCTL) &
457  ~SYSCTL_TMR2CLKCTL_TMR2CLKPRESCALE_M) |
458  ((uint16_t)prescaler <<SYSCTL_TMR2CLKCTL_TMR2CLKPRESCALE_S);
459 
460  EDIS;
461  }
462  SYSCTL_REGWRITE_DELAY;
463 }
