375 //*****************************************************************************
376 //
382 //
383 //*****************************************************************************
384 static inline DCSM_SecurityStatus
385 DCSM_getZone1CSMSecurityStatus(void)
386 {
387  uint32_t status;
388  DCSM_SecurityStatus returnStatus;
389  status = HWREG(DCSM_Z1_BASE + DCSM_O_Z1_CR);
390 
391  //
392  // if ARMED bit is set and UNSECURED bit is set then CSM is unsecured.
393  // Else it is secure.
394  //
395  if(((status & DCSM_Z1_CR_ARMED) != 0U) &&
396  ((status & DCSM_Z1_CR_UNSECURE) != 0U))
397  {
398  returnStatus = DCSM_STATUS_UNSECURE;
399  }
400  else if((status & DCSM_Z1_CR_ALLONE) == DCSM_Z1_CR_ALLONE)
401  {
402  returnStatus = DCSM_STATUS_BLOCKED;
403  }
404  else if((status & DCSM_Z1_CR_ALLZERO) == DCSM_Z1_CR_ALLZERO)
405  {
406  returnStatus = DCSM_STATUS_LOCKED;
407  }
408  else
409  {
410  returnStatus = DCSM_STATUS_SECURE;
411  }
412 
413  return(returnStatus);
414 }

541 //*****************************************************************************
542 //
553 //
554 //*****************************************************************************
555 static inline DCSM_MemoryStatus
556 DCSM_getFlashSectorZone(DCSM_Sector sector)
557 {
558  uint32_t sectStat;
559  uint16_t shift;
560 
561  //
562  // Get the Sector status register for the specific bank
563  //
564  if(sector <= DCSM_BANK1_SECTOR_127_96)
565  {
566  sectStat = HWREG(DCSMCOMMON_BASE + DCSM_O_SECTSTAT1);
567  shift = (uint16_t)sector * 2U;
568  }
569  else if(sector <= DCSM_BANK3_SECTOR_127_96)
570  {
571  sectStat = HWREG(DCSMCOMMON_BASE + DCSM_O_SECTSTAT2);
572  shift = ((uint16_t)sector & 0xFU) * (uint16_t)2U;
573  }
574  else
575  {
576  sectStat = HWREG(DCSMCOMMON_BASE + DCSM_O_SECTSTAT3);
577  shift = ((uint16_t)sector & 0xFU) * (uint16_t)2U;
578  }
579 
580  //
581  //Read the SECTSTAT register for the specific Flash Sector.
582  //
583  return((DCSM_MemoryStatus)((uint16_t)((sectStat >> shift) & 0x3U)));
584 }

700 //*****************************************************************************
701 static inline uint32_t
702 DCSM_getZone1OTPSecureLockStatus(DCSM_OTPLock lockType)
703 {
704  uint32_t status, returnStatus;
705  status = HWREG(DCSM_Z1_BASE + DCSM_O_Z1_OTPSECLOCK);
706 
707  //
708  // Reflects the state of the OTP Sec LOCK feature.
709  //
710  if(lockType == DCSM_OTPSECLOCK_JTAG)
711  {
712  returnStatus = (status & DCSM_Z1_OTPSECLOCK_JTAGLOCK_M) >>
713  DCSM_Z1_OTPSECLOCK_JTAGLOCK_S;
714  }
715  else if(lockType == DCSM_OTPSECLOCK_CRCLOCK)
716  {
717  returnStatus = (status & DCSM_Z1_OTPSECLOCK_CRCLOCK_M) >>
718  DCSM_Z1_OTPSECLOCK_CRCLOCK_S;
719  }
720  else if(lockType == DCSM_OTPSECLOCK_PSWDLOCK)
721  {
722  returnStatus = (status & DCSM_Z1_OTPSECLOCK_PSWDLOCK_M) >>
723  DCSM_Z1_OTPSECLOCK_PSWDLOCK_S;
724  }
725  else
726  {
727  returnStatus = (uint32_t)DCSM_OTPSECLOCK_INVALID;
728  }
729 
730  return(returnStatus);
731 }
