341 //*****************************************************************************
342 //
354 //
355 //*****************************************************************************
356 static inline void ECAP_setEventPrescaler(uint32_t base,
357  uint16_t preScalerValue)
358 {
359  ASSERT(ECAP_isBaseValid(base));
360 
361  ASSERT(preScalerValue < ECAP_MAX_PRESCALER_VALUE);
362 
363  EALLOW;
364 
365  //
366  // Write to PRESCALE bit
367  //
368  HWREGH(base + ECAP_O_ECCTL1) =
369  ((HWREGH(base + ECAP_O_ECCTL1) & (~ECAP_ECCTL1_PRESCALE_M)) |
370  (preScalerValue << ECAP_ECCTL1_PRESCALE_S));
371  EDIS;
372 }

374 //*****************************************************************************
375 //
390 //
391 //*****************************************************************************
392 static inline void ECAP_setEventPolarity(uint32_t base,
393  ECAP_Events event,
394  ECAP_EventPolarity polarity)
395 {
396 
397  uint16_t shift;
398 
399  ASSERT(ECAP_isBaseValid(base));
400 
401  shift = ((uint16_t)event) << 1U;
402 
403  EALLOW;
404 
405  //
406  // Write to CAP1POL, CAP2POL, CAP3POL or CAP4POL
407  //
408  HWREGH(base + ECAP_O_ECCTL1) =
409  (HWREGH(base + ECAP_O_ECCTL1) & ~(1U << shift)) |
410  ((uint16_t)polarity << shift);
411  EDIS;
412 }

504 //*****************************************************************************
505 static inline void ECAP_enableInterrupt(uint32_t base,
506  uint16_t intFlags)
507 {
508  ASSERT(ECAP_isBaseValid(base));
509  ASSERT((intFlags & ~(ECAP_ISR_SOURCE_CAPTURE_EVENT_1 |
510  ECAP_ISR_SOURCE_CAPTURE_EVENT_2 |
511  ECAP_ISR_SOURCE_CAPTURE_EVENT_3 |
512  ECAP_ISR_SOURCE_CAPTURE_EVENT_4 |
513  ECAP_ISR_SOURCE_COUNTER_OVERFLOW |
514  ECAP_ISR_SOURCE_COUNTER_PERIOD |
515  ECAP_ISR_SOURCE_COUNTER_COMPARE)) == 0U);
516 
517 
518  EALLOW;
519 
520  //
521  // Set bits in ECEINT register
522  //
523  HWREGH(base + ECAP_O_ECEINT) |= intFlags;
524  EDIS;
525 }
