446 //*****************************************************************************
447 //
461 //
462 //*****************************************************************************
463 static inline void
464 EPG_selectEPGOutput(uint32_t base, EPG_OUT gPinNum,
465  EPG_OutputSelect gPinOutSel)
466 {
467  uint32_t regValue, bitFieldLoc, bitFieldVal;
468 
469  ASSERT(EPG_isBaseValid(base));
470  ASSERT((uint16_t)gPinNum <= 7U);
471 
472  bitFieldLoc = 1UL << ((uint16_t)gPinNum + EPG_GCTL0_EPGOUTSEL_S);
473  bitFieldVal = (uint32_t)gPinOutSel << ((uint16_t)gPinNum +
474  EPG_GCTL0_EPGOUTSEL_S);
475 
476  regValue = (HWREG(base + EPG_O_GCTL0) & (~bitFieldLoc));
477  regValue |= bitFieldVal;
478 
479  HWREG((base + EPG_O_GCTL0)) = regValue;
480 }
