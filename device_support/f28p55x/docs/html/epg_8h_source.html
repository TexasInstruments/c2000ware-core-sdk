432 //*****************************************************************************
433 //
447 //
448 //*****************************************************************************
449 static inline void
450 EPG_selectEPGOutput(uint32_t base, EPG_OUT gPinNum,
451  EPG_OutputSelect gPinOutSel)
452 {
453  uint32_t regValue, bitFieldLoc, bitFieldVal;
454 
455  ASSERT(EPG_isBaseValid(base));
456  ASSERT((uint16_t)gPinNum <= 7U);
457 
458  bitFieldLoc = 1UL << ((uint16_t)gPinNum + EPG_GCTL0_EPGOUTSEL_S);
459  bitFieldVal = (uint32_t)gPinOutSel << ((uint16_t)gPinNum +
460  EPG_GCTL0_EPGOUTSEL_S);
461 
462  regValue = (HWREG(base + EPG_O_GCTL0) & (~bitFieldLoc));
463  regValue |= bitFieldVal;
464 
465  HWREG((base + EPG_O_GCTL0)) = regValue;
466 }
