165 static inline void
166 Flash_setWaitstates(uint32_t ctrlBase, uint16_t waitstates)
167 {
168  //
169  // Check the arguments.
170  //
171  ASSERT(Flash_isCtrlBaseValid(ctrlBase));
172 
173  //
174  // waitstates is 4 bits wide.
175  //
176  ASSERT(waitstates <= 0xFU);
177 
178  EALLOW;
179  //
180  // Write flash read wait-state amount to appropriate register.
181  //
182  HWREG(ctrlBase + FLASH_O_FRDCNTL) =
183  (HWREG(ctrlBase + FLASH_O_FRDCNTL) &
184  ~(uint32_t)FLASH_FRDCNTL_RWAIT_M) |
185  ((uint32_t)waitstates << FLASH_FRDCNTL_RWAIT_S);
186  EDIS;
187 }
