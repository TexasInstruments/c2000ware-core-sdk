421 static inline uint32_t
422 GPIO_readPin(uint32_t pin)
423 {
424  volatile uint32_t *gpioDataReg;
425 
426  //
427  // Check the arguments.
428  //
429  ASSERT(GPIO_isPinValid(pin));
430 
431  gpioDataReg = (uint32_t *)((uintptr_t)GPIODATA_BASE) +
432  ((pin / 32U) * GPIO_DATA_REGS_STEP);
433 
434  return((gpioDataReg[GPIO_GPxDAT_INDEX] >> (pin % 32U)) & (uint32_t)0x1U);
435 }

486 static inline void
487 GPIO_writePin(uint32_t pin, uint32_t outVal)
488 {
489  volatile uint32_t *gpioDataReg;
490  uint32_t pinMask;
491 
492  //
493  // Check the arguments.
494  //
495  ASSERT(GPIO_isPinValid(pin));
496 
497  gpioDataReg = (uint32_t *)((uintptr_t)GPIODATA_BASE) +
498  ((pin / 32U) * GPIO_DATA_REGS_STEP);
499 
500  pinMask = (uint32_t)1U << (pin % 32U);
501 
502  if(outVal == 0U)
503  {
504  gpioDataReg[GPIO_GPxCLEAR_INDEX] = pinMask;
505  }
506  else
507  {
508  gpioDataReg[GPIO_GPxSET_INDEX] = pinMask;
509  }
510 }
