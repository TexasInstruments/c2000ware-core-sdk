745 static inline void
746 HRPWM_setMEPEdgeSelect(uint32_t base, HRPWM_Channel channel,
747  HRPWM_MEPEdgeMode mepEdgeMode)
748 {
749  //
750  // Check the arguments
751  //
752  ASSERT(HRPWM_isBaseValid(base));
753 
754  //
755  // Set the edge mode
756  //
757  EALLOW;
758  HWREGH(base + HRPWM_O_HRCNFG) =
759  ((HWREGH(base + HRPWM_O_HRCNFG) & ~(0x3U << (uint16_t)channel )) |
760  ((uint16_t)mepEdgeMode << (uint16_t)channel));
761  EDIS;
762 }
