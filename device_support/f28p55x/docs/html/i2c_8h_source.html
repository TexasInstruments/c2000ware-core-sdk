427 //*****************************************************************************
428 //
446 //
447 //*****************************************************************************
448 static inline void
449 I2C_setFIFOInterruptLevel(uint32_t base, I2C_TxFIFOLevel txLevel,
450  I2C_RxFIFOLevel rxLevel)
451 {
452  //
453  // Check the arguments.
454  //
455  ASSERT(I2C_isBaseValid(base));
456 
457  //
458  // Set the FIFO interrupt levels.
459  //
460  HWREGH(base + I2C_O_FFTX) = (HWREGH(base + I2C_O_FFTX) &
461  (~I2C_FFTX_TXFFIL_M)) | (uint16_t)txLevel;
462  HWREGH(base + I2C_O_FFRX) = (HWREGH(base + I2C_O_FFRX) &
463  (~I2C_FFRX_RXFFIL_M)) | (uint16_t)rxLevel;
464 }

485 //
486 //*****************************************************************************
487 static inline void
488 I2C_getFIFOInterruptLevel(uint32_t base, I2C_TxFIFOLevel *txLevel,
489  I2C_RxFIFOLevel *rxLevel)
490 {
491  //
492  // Check the arguments.
493  //
494  ASSERT(I2C_isBaseValid(base));
495 
496  //
497  // Extract the transmit and receive FIFO levels.
498  //
499  *txLevel = (I2C_TxFIFOLevel)(HWREGH(base + I2C_O_FFTX) &
500  I2C_FFTX_TXFFIL_M);
501  *rxLevel = (I2C_RxFIFOLevel)(HWREGH(base + I2C_O_FFRX) &
502  I2C_FFRX_RXFFIL_M);
503 }
