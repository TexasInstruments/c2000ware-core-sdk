244 static inline void
245 Interrupt_register(uint32_t interruptNumber, void (*handler)(void))
246 {
247  uint32_t address;
248 
249  //
250  // Calculate appropriate address for the interrupt number
251  //
252  address = (uint32_t)PIEVECTTABLE_BASE +
253  (((interruptNumber & 0xFFFF0000U) >> 16U) * 2U);
254 
255  //
256  // Copy ISR address into PIE table
257  //
258  EALLOW;
259  HWREG(address) = (uint32_t)handler;
260  EDIS;
261 }
