327 static inline void
328 MemCfg_setCLAMemType(uint32_t ramSections, MemCfg_CLAMemoryType claMemType)
329 {
330  //
331  // Check the arguments.
332  //
333  ASSERT((ramSections & MEMCFG_SECT_TYPE_MASK) == MEMCFG_SECT_TYPE_LS);
334 
335  //
336  // Write the CLA memory configuration to the appropriate register. Either
337  // set or clear the bit that determines the function of the RAM section as
338  // it relates to the CLA.
339  //
340  EALLOW;
341 
342  if(claMemType == MEMCFG_CLA_MEM_PROGRAM)
343  {
344  //
345  // Program memory
346  //
347  HWREG(MEMCFG_BASE + MEMCFG_O_LSXCLAPGM) |= ramSections;
348  }
349  else
350  {
351  //
352  // Data memory
353  //
354  HWREG(MEMCFG_BASE + MEMCFG_O_LSXCLAPGM) &= ~ramSections;
355  }
356 
357  EDIS;
358 }

386 static inline void
387 MemCfg_enableViolationInterrupt(uint32_t intFlags)
388 {
389  //
390  // Enable the specified interrupts.
391  //
392  EALLOW;
393 
394  HWREG(ACCESSPROTECTION_BASE + MEMCFG_O_NMAVINTEN) |=
395  intFlags & MEMCFG_NMVIOL_MASK;
396 
397  HWREG(ACCESSPROTECTION_BASE + MEMCFG_O_MAVINTEN) |=
398  (intFlags & MEMCFG_MVIOL_MASK) >> MEMCFG_MVIOL_SHIFT;
399 
400  EDIS;
401 }

404 //

432 static inline void
433 MemCfg_disableViolationInterrupt(uint32_t intFlags)
434 {
435  //
436  // Disable the specified interrupts.
437  //
438  EALLOW;
439 
440  HWREG(ACCESSPROTECTION_BASE + MEMCFG_O_NMAVINTEN) &=
441  ~(intFlags & MEMCFG_NMVIOL_MASK);
442 
443  HWREG(ACCESSPROTECTION_BASE + MEMCFG_O_MAVINTEN) &=
444  ~((intFlags & MEMCFG_MVIOL_MASK) >> MEMCFG_MVIOL_SHIFT);
445 
446  EDIS;
447 }

473 //*****************************************************************************
474 static inline uint32_t
475 MemCfg_getViolationInterruptStatus(void)
476 {
477  uint32_t status;
478 
479  //
480  // Read and return RAM access status flags.
481  //
482  status = (HWREG(ACCESSPROTECTION_BASE + MEMCFG_O_NMAVFLG)) |
483  (HWREG(ACCESSPROTECTION_BASE + MEMCFG_O_MAVFLG) <<
484  MEMCFG_MVIOL_SHIFT);
485 
486  return(status);
487 }

517 MemCfg_forceViolationInterrupt(uint32_t intFlags)
518 {
519  //
520  // Shift and mask the flags appropriately and write them to the
521  // corresponding SET register.

526  intFlags & MEMCFG_NMVIOL_MASK;

529  (intFlags & MEMCFG_MVIOL_MASK) >> MEMCFG_MVIOL_SHIFT;
530 
531  EDIS;
